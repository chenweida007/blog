function myFindIndex(arr, val) {
	let low = 0;
	let high = arr.length - 1;
	while(low < high) {
		let mid = parseInt(low + (high - low)/2);
		if(val == arr[mid]) {
			return mid
		}
		if(val < arr[mid]) {
			high = mid
		}
		if(val > arr[mid]) {
			low = mid
		}
	}
	return -1
}
